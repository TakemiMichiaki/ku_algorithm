\begin{algorithm}
% \caption{INSERTION-SORT($A$)}
 INSERTION-SORT($A$)
 \begin{algorithmic}[1]
  \NoDo
  \For{$j = 2$}\TO{$A.length$}
   \State $key = A[j]$
   \State // Insert $A[j]$ into the sorted sequence $A[1 \, .. \, j-1]$.
   \State $i = j - 1$
   \While{$i > 0$ and $A[i] > key$}
    \State $A[i+1] = A[i]$
    \State $i = i - 1$
   \EndWhile
   \State $A[i+1] = key$
  \EndFor
 \end{algorithmic}
\end{algorithm}

\begin{algorithm}
 LINEAR-SEARCH($A, v$)
 \begin{algorithmic}[1]
  \State $i =$ NIL
  \NoDo
  \For{$j = 1$}\TO{$A.length$}
   \NoThen
   \If{$A[j] == v$}
    \State $i = j$
    \State \RETURN{$i$}
   \EndIf
  \EndFor
  \State \RETURN{NIL}
 \end{algorithmic}
\end{algorithm}
